204      private void playAnim(final View view, final int value, final String data){
205          view.animate().alpha(value).scaleX(value).scaleY(value).setDuration(500).setStartDelay(100)
206                  .setInterpolator(new DecelerateInterpolator()).setListener(new Animator.AnimatorListener() {
207              @Override
208              public void onAnimationStart(Animator animation) {
209                  if(value == 0 && count < 4){
210                      String option = "";
211                      if(count == 0){
212  
213                          option = list.get(position).getOptionA();
214                      }else if(count == 1){
215  
216                          option = list.get(position).getOptionB();
217                      }else if(count == 2){
218  
219                          option = list.get(position).getOptionC();
220                      }else if(count == 3){
221  
222                          option = list.get(position).getOptionD();
223                      }
224                      playAnim(optionsContainer.getChildAt(count), 0, option);
225                      count ++;
226                  }
227              }
228  
229              @Override
230              public void onAnimationEnd(Animator animation) {
231                  if(value == 0){
232                      try{
233                          ((TextView) view).setText(data);
234                          noIndicator.setText(position+1+"/"+list.size());
235  
236                          if(modelMatch()){
237                              bookmarkbtn.setImageDrawable(getDrawable(R.drawable.bookmark));
238                          }else{
239                              bookmarkbtn.setImageDrawable(getDrawable(R.drawable.bookmark_border));
240                          }
241  
242                      }catch (ClassCastException ex){
243                          ((Button) view).setText(data);
244  
245                      }
246                      view.setTag(data);
247                      playAnim(view,1, data);
248                  }
249              }
250  
251              @Override
252              public void onAnimationCancel(Animator animation) {
253  
254              }
255  
256              @Override
257              public void onAnimationRepeat(Animator animation) {
258  
259              }
260          });
261      }

263      private void checkAnswer(Button selectedOption){
264          enableOption(false);
265          nextBtn.setEnabled(true);
266          nextBtn.setAlpha(1);
267  
268          if(selectedOption.getText().toString().equals(list.get(position).getCorrectANS())){
269              // correct
270              score++;
271              selectedOption.setBackgroundTintList(ColorStateList.valueOf(Color.parseColor("#4CAF50")));
272  
273          }else{
274              // incorrect
275              selectedOption.setBackgroundTintList(ColorStateList.valueOf(Color.parseColor("#FF0000")));
276              Button correctoption = (Button) optionsContainer.findViewWithTag(list.get(position).getCorrectANS());
277              correctoption.setBackgroundTintList(ColorStateList.valueOf(Color.parseColor("#4CAF50")));
278          }
279      }

283              optionsContainer.getChildAt(i).setEnabled(enabled);
284              if (enabled){

286              }
287          }

290      private void getBookmarks(){
291  
292          String json = preferences.getString(KEY_NAME, "");
293  
294          Type type = new TypeToken<List<QuestionModel>>(){}.getType();
295  
296          bookmarksList = gson.fromJson(json, type);
297  
298          if(bookmarksList == null){
299              bookmarksList = new ArrayList<>();
300          }
301      }
302  
303      private boolean modelMatch(){
304          boolean matched = false;
305          int i =0;
306  
307          for(QuestionModel model : bookmarksList){
308  
309              if(model.getQuestion().equals(list.get(position).getQuestion())
310              && model.getCorrectANS().equals(list.get(position).getCorrectANS())
311              && model.getSetNo() == list.get(position).getSetNo()){
312  
313                  matched = true;
314  
315                  matchedQuestionPosition = i;
316              }
317              i++;
318          }
319          return matched;
320      }
321  
322      private void storeBookmarks(){
323  
324          String json = gson.toJson(bookmarksList);
325  
326          editor.putString(KEY_NAME, json);
327          editor.commit();
328      }

330      public void loadAds(){
331          AdView mAdView = findViewById(R.id.adView);
332          AdRequest adRequest = new AdRequest.Builder().build();
333          mAdView.loadAd(adRequest);
334  
335          interstitialAd = new InterstitialAd(this);
336          interstitialAd.setAdUnitId(getResources().getString(R.string.interstitialAd_id));
337          interstitialAd.loadAd(new AdRequest.Builder().build());
338          interstitialAd.setAdListener(new AdListener(){
339              @Override
340              public void onAdClosed() {
341                  super.onAdClosed();
342                  interstitialAd.loadAd(new AdRequest.Builder().build());
343  
344                  Intent scoreIntent = new Intent(QuestionsActivity.this, ScoreActivity.class);
345                  scoreIntent.putExtra("score", score);
346                  scoreIntent.putExtra("total", list.size());
347                  startActivity(scoreIntent);
348                  finish();
349                  return;
350              }
351          });
352      }
