38       @Override
39       protected void onCreate(Bundle savedInstanceState) {
40           super.onCreate(savedInstanceState);
41           setContentView(R.layout.activity_categories);
42   
43           Toolbar toolbar = findViewById(R.id.toolbar);
44   
45           loadAds();
46   
47           setSupportActionBar(toolbar);
48           getSupportActionBar().setTitle("Categories");
49           getSupportActionBar().setDisplayHomeAsUpEnabled(true);// back button
50   
51           loadingDialog = new Dialog(this);
52           loadingDialog.setContentView(R.layout.loading);
53           loadingDialog.getWindow().setBackgroundDrawable(getDrawable(R.drawable.rounded_corners));
54           loadingDialog.getWindow().setLayout(LinearLayout.LayoutParams.WRAP_CONTENT, LinearLayout.LayoutParams.WRAP_CONTENT);
55           loadingDialog.setCancelable(false);
56   
57           recyclerView = findViewById(R.id.rv);
58   
59           LinearLayoutManager layoutManager = new LinearLayoutManager(this);

64           list = new ArrayList<>();
65   
66   
67   
68          final CategoryAdapter adapter = new CategoryAdapter(list);
69          recyclerView.setAdapter(adapter);
70   
71          loadingDialog.show();
72          myRef.child("Categories").addListenerForSingleValueEvent(new ValueEventListener() {
73              @Override
74              public void onDataChange(@NonNull DataSnapshot dataSnapshot) {
75   
76                  for (DataSnapshot dataSnapshot1 : dataSnapshot.getChildren()){
77                      list.add(dataSnapshot1.getValue(CategoryModel.class));
78                  }
79   
80                  adapter.notifyDataSetChanged();
81                  loadingDialog.dismiss();
82   
83              }
84   
85              @Override
86              public void onCancelled(@NonNull DatabaseError databaseError) {
87                  Toast.makeText(CategoriesActivity.this, databaseError.getMessage(), Toast.LENGTH_SHORT).show();
88                  loadingDialog.dismiss();
89                  finish();
90              }
91          });
92   
93       }

100          if(item.getItemId() == android.R.id.home){
